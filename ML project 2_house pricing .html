#remove outlier
def remove_outlier(d,c):
    q1=d[c].quantile(0.25)
    q2=d[c].quantile(0.75)
    
    iqr=q2-q1
    
    lb=q1-1.5*iqr
    ub=q2+1.5*iqr
    
    data_without_outlier=d[(d[c]<=ub) & (d[c]>=lb)]
    return data_without_outlier
